# Common infrastructure
# ---------------------

RUN apt-get update -q && \
    apt-get install -q -y --no-install-recommends make curl && \
    apt-get clean && rm -rf /var/lib/apt/lists/*

RUN pip install jinja2
RUN pip install prospector[with_everything]
RUN pip install pytest pytest-cov pytest-benchmark[histogram] isort[colors] black hypothesis
RUN pip install poetry
RUN pip install build

# Project
# -------

COPY . /opt/{{ project_name }}
WORKDIR /opt/{{ project_name }}
RUN make clean
RUN make buildingbits
RUN make checks
RUN make format

# Project install procedure (pyproject.toml may have to be altered based on the way the project is built/installed)
# -------------------------
RUN python setup.py install

RUN make tests_
